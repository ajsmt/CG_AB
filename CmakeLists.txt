cmake_minimum_required(VERSION 3.15)
project(ComputerGraphicsLabs CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


add_library(imgui INTERFACE)
file(GLOB IMGUI_SOURCES
    "libs/imgui/*.cpp"
    "libs/imgui/backends/imgui_impl_glfw.cpp"
    "libs/imgui/backends/imgui_impl_opengl3.cpp"
)

target_sources(imgui INTERFACE ${IMGUI_SOURCES})

target_include_directories(imgui INTERFACE
    "libs/imgui"
    "libs/imgui/backends"
)

add_library(glfw INTERFACE)
target_include_directories(glfw INTERFACE "libs/glfw/include")
target_link_directories(glfw INTERFACE "libs/glfw/lib-mingw-w64")

find_package(OpenGL REQUIRED)

target_link_libraries(glfw INTERFACE
    glfw3
    gdi32
    ${OPENGL_LIBRARIES}
    imm32
)

add_subdirectory(Lab_1)
